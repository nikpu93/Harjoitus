CREATE TABLE Planeetta(
ID INT,
Nimi VARCHAR(20) NOT NULL,
Vakiluku VARCHAR(20) NOT NULL,
Url VARCHAR(50) NOT NULL,
PRIMARY KEY(ID));
CREATE TABLE Ihmiset(
ID INT,
Nimi VARCHAR(20) NOT NULL,
Syntynyt VARCHAR(20) NOT NULL,
Url VARCHAR(50) NOT NULL,
PRIMARY KEY(ID));
CREATE TABLE Lajit(
ID INT,
Nimi VARCHAR(20) NOT NULL,
Kotiplaneetta VARCHAR(50) NOT NULL,
Url VARCHAR(50) NOT NULL,
PRIMARY KEY(ID));
